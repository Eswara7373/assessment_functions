7.Write a function that accepts parameter n and returns factorial of n

function factorial(n){
        let answer = 1;
        if (n == 0 || n == 1){
          console.log(answer);
        }else{
          for(var i = n; i >= 1; i--){
            answer = answer * i;
          }
          console.log(answer);
        }  
      }
       factorial(4);
